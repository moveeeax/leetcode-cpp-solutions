# **Binary Tree Zigzag Level Order Traversal**  

## **1️⃣ Problem Understanding**  
### **📌 Intuition**  
The task here is to traverse a binary tree in a zigzag manner, alternating the order of traversal between left-to-right and right-to-left for each level. This requires us to keep track of the current level and change the order of the traversal based on whether we're at an even or odd level.

### **🚀 Approach**  
1. **Use of Queue**: We can utilize a queue to traverse the tree level by level. Each node's children can be added to the queue for the next level traversal.
2. **Direction Toggle**: We will use a boolean flag that determines whether the current level will be traversed left to right or right to left.
3. **Store Results**: We'll store the values of each level in a temporary list, and once a level is fully processed, we will store this list in the final result vector.

### **⏱️ Complexity Analysis**  
- **Time Complexity**: O(n), where n is the number of nodes in the binary tree since we visit each node exactly once.  
- **Space Complexity**: O(n) for the queue to hold nodes at the current level.

## **2️⃣ LeetCode Submission**  
```cpp
class Solution {
public:
    vector<vector<int>> zigzagLevelOrder(TreeNode* root) {
        vector<vector<int>> result;
        if (!root) return result;

        queue<TreeNode*> q;
        q.push(root);
        bool leftToRight = true;

        while (!q.empty()) {
            int levelSize = q.size();
            vector<int> levelValues(levelSize);
            for (int i = 0; i < levelSize; i++) {
                TreeNode* currentNode = q.front();
                q.pop();
                // Find position for current node value
                int index = leftToRight ? i : (levelSize - 1 - i);
                levelValues[index] = currentNode->val;

                // Add children to the queue
                if (currentNode->left) q.push(currentNode->left);
                if (currentNode->right) q.push(currentNode->right);
            }
            result.push_back(levelValues);
            leftToRight = !leftToRight; // Toggle the direction
        }
        return result;
    }
};
```

#include <iostream>
#include <vector>
#include <queue>

using namespace std;

// Definition for a binary tree node.
struct TreeNode {
    int val;
    TreeNode *left;
    TreeNode *right;
    TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
};

class Solution {
public:
    vector<vector<int>> zigzagLevelOrder(TreeNode* root) {
        vector<vector<int>> result;
        if (!root) return result;

        queue<TreeNode*> q;
        q.push(root);
        bool leftToRight = true;

        while (!q.empty()) {
            int levelSize = q.size();
            vector<int> levelValues(levelSize);
            for (int i = 0; i < levelSize; i++) {
                TreeNode* currentNode = q.front();
                q.pop();
                // Find position for current node value
                int index = leftToRight ? i : (levelSize - 1 - i);
                levelValues[index] = currentNode->val;

                // Add children to the queue
                if (currentNode->left) q.push(currentNode->left);
                if (currentNode->right) q.push(currentNode->right);
            }
            result.push_back(levelValues);
            leftToRight = !leftToRight; // Toggle the direction
        }
        return result;
    }
};

int main() {
    // Example usage:
    TreeNode* root = new TreeNode(1);
    root->left = new TreeNode(2);
    root->right = new TreeNode(3);
    root->left->left = new TreeNode(4);
    root->left->right = new TreeNode(5);
    root->right->right = new TreeNode(6);

    Solution solution;
    vector<vector<int>> result = solution.zigzagLevelOrder(root);

    for (const auto& level : result) {
        for (int value : level) {
            cout << value << " ";
        }
        cout << endl;
    }

    return 0;
}
